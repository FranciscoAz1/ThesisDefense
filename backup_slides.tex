\begin{frame}{Failure Modes}
\begin{itemize}
  \item Sparse graph: limited hops \rightarrow fallback vector only
  \item Conflicting updates without timestamps: unresolved contradiction
  \item Entity over-merging: conflated contexts
  \item Long-tail metadata: low embedding recall
\end{itemize}
\end{frame}
